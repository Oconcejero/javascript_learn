let primerProducto = document.querySelector(".producto");

//Primera propiedad
console.log(primerProducto.getAttribute("data-price"));
//23.5

//Segunda propiedad
primerProducto.dataset
//DOMStringMap {id:'1', price:'23.5'}

primerProducto.dataset.price //23.5

//Cambiar el id
primerProducto.dataset.id = '5' //class="producto" data-id="5" data-price="23.5">

//Añadir otro atributo
primerProducto.database.tax = 'true' //class="producto" data-id="5" data-price="23.5" data-tax='true'>
  
//Eliminar el atributo
primerProducto.removeAttribute('data-tax') //class="producto" data-id="5" data-price="23.5">
