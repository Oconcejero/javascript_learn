Arrow functions


  // Se sustituye "function" por una flecha
  // y se coloca después
  const saludar = ()=>{

  }

  const saludar = (name) => {
      console.log(`¡Hola, ${name}!`)
  }

  //Forma de simplificarla más si es simple si tiene
  //un parámetro solo y si solo tiene una expresión
  const saludar2 = (name) => console.log(`¡Hola, ${name}!`)

  saludar("Luis") // ¡Hola, Luis!
  saludar2("Enrique") // ¡Hola, Enrique!

Parámetros


  function sum(a, b) {
      console.log(a + b)
  }

  sum(5, 10)
  sum(5) //Nan, error al sumar 5 y nada
  sum() //Nan, error al no sumar ningún parámetro

  //Manera de asignar "por defecto" el
  //valor 0 a los parámetros y evita errores
  function defaultSum(a = 0, b = 0) {
      console.log(a + b)
  }

Por defecto


  defaultSum()
  defaultSum(5)
  defaultSum(5, 10)
  defaultSum(undefined, 5)

forEach


  //Función que nos sirve para ejecutar bucles
  //a elementos iterables
  const myArray = [1, 2, 3, 4]

  const mySet = new Set(["Brais", "Moure", "mouredev", 37, true, "braismoure@mouredev.com"])

  const myMap = new Map([
      ["name", "Brais"],
      ["email", "braismoure@mouredev.com"],
      ["age", 37]
  ])

  myArray.forEach(function (value) {
      console.log(value)
  })

  myArray.forEach((value) => console.log(value))

  mySet.forEach((value) => console.log(value))

  myMap.forEach((value) => console.log(value))
