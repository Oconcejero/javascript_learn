OBJETO MATH

RETURN

sqrt( )


  //Devuelve la raíz cuadrada positiva de un número
  
  let numero = Math.sqrt(25)
  console.log(numero) //5, La raíz cuadrada de 25

cbrt( )


  //Devuelve la raíz cúbica de un número

  let numero = Math.cbrt(25)
  console.log(numero) //2.924017738212866, La raíz cúbica de 25

max( ) & min( )


  //Devuelven el mayor de cero o más números y
  //el más pequeño de cero o más números

  let numero = Math.max(4,5,7,8,6,25,35,124,584,21,5)
  console.log(numero) //584, Devuelve el más grande

  let numero = Math.min(4,5,7,8,6,25,35,124,584,21,5)
  console.log(numero) //4, Devuelve el más pequeño

random( )


  //Devuelve un número aleatorio entre 0 y 1

  let numero = Math.random()
  console.log(numero) //0.16092775158390604

round( )


  //Devuelve el valor de un número redondeado al número entero más cercano

  let numero = Math.random()*100;
  numero = Math.round(numero);
  console.log(numero) //74

fround( )


  //Devuelve la representación flotante de precisión simple
  //más cercana de un número

  let numero = Math.fround(9.45215478542513265);
  console.log(numero) //9.452155113220215 redondea a 4 bytes

floor( )


  //Devuelve el número entero redondeado hacia abajo
  
  let numero = Math.floor(3.999);
  console.log(numero) //3

trunc( )


  //Devuelve la parte entera del número x, la eliminación de los dígitos fraccionarios
 
  let numero = Math.trunc(9.10);
  console.log(numero) //9 Elimina los decimales

Propiedades


  //PI
