Métodos

myArray = []

push( ) y pop( )


  //Añade elementos al final de la lista
  
  let nombre = ['Pedro','María','Jorge']
  nombre.push("Luis")
  console.log(nombre) // [ 'Pedro', 'María', 'Jorge', 'Luis' ]
  
  // Elimina el último elemento y lo retorna
  
  let nombre = ['Pedro','María','Jorge']
  let resultado = nombre.pop()
  console.log(resultado) // Jorge
  console.log(nombre) // [ 'Pedro', 'María' ]

shift( ) y unshift( )


  //Elimina el primer elemento de la lista
  
  let nombre = ['Pedro','María','Jorge']
  let resultado = nombre.shift()
  console.log(resultado) // Pedro
  console.log(nombre) // [ 'María', 'Jorge' ]

  //Añade los elementos al principio de la lista
  
  let nombre = ['Pedro','María','Jorge']
  nombre.unshift("Luis")
  console.log(nombre) // [ 'Luis', 'Pedro', 'María', 'Jorge' ]

reverse( )


  //Invierte el orden de los elementos de un array
  
  let nombre = ['Pedro','María','Jorge']
  let resultado = nombre.reverse()
  console.log(resultado) // [ 'Jorge', 'María', 'Pedro' ]

sort( )


  //Ordena los elementos y los devuelve ordenados
  
  let numeros = [2,5,9,4,3,1]
  let resultado = numeros.sort()
  console.log(resultado) // [ 1, 2, 3, 4, 5, 9 ]

length


  let numeros = [2,5,9,4,3,1]
  let resultado = numeros.length
  console.log(resultado) // 6

Clear


  myArray = []
  myArray.length = 0 // alternativa
  console.log(myArray)

slice( )


  //Devuelve una parte del array dentro de un nuevo array
  //empezando por el inicio hasta el fin (fin no incluido)
  
  let nombre = ['Manzana','Pera','Melón','Sandía']
  let resultado = nombre.slice(0,2)
  console.log(resultado) // [ 'Manzana', 'Pera' ]

  //Para seleccionar todos menos el último sin saber cuantos hay
  
  let nombre = ['Manzana','Pera','Melón','Sandía']
  let resultado = nombre.slice(0,-1)
  console.log(resultado) // [ 'Manzana', 'Pera', 'Melón' ]

splice( )


  //Remueve elementos según la posición indicada

  let numeros = [1,2,3,4,5,6]
  numeros.splice(1,3)
  console.log(numeros) // [ 1, 5, 6 ]

  //También puede agregar elementos según la posición indicada

  let numeros = ['Carmen','Paco','Matilde','Enma','Fer','Miriam','Jazmín']
  numeros.splice(1,3,"Luis","Ramón")
  console.log(numeros) // [ 'Carmen', 'Luis', 'Ramón', 'Fer', 'Miriam', 'Jazmín' ]

join( )


  //Une todos los elementos de una matris en una cadena y la devuelve
  //Pudiendo usar un separador específico

  let nombre = ['Pedro','María','Jorge']
  let resultado = nombre.join()
  console.log(resultado) // Pedro,María,Jorge

  let nombre = ['Pedro','María','Jorge']
  let resultado = nombre.join(" - ")
  console.log(resultado) // Pedro - María - Jorge

  //Se puede considerar un bucle
  
  let nombres = ['Pedro','María','Jorge','Pepa','Antonio']

  //Cuando cumpla una condición
  let nombres = ['Pedro','María','Jorge','Pepa','Antonio']
  let resultado = nombres.filter(nombre => nombre.length > 4)
  console.log(resultado) //[ 'Pedro', 'María', 'Jorge', 'Antonio' ]
