Push y Pop


  //Añade elementos al final de la lista
  myArray.push("Brais")
  myArray.push("Moure")
  myArray.push("mouredev")
  myArray.push(37)

  console.log(myArray)
  
  // Elimina el último y lo devuelve
  console.log(myArray.pop()) 
  myArray.pop()

  console.log(myArray)

Shift y Unshift


  //Elimina el primer elemento de la lista
  console.log(myArray.shift())
  console.log(myArray)

  //Añade los elementos al principio de la lista
  myArray.unshift("Brais", "mouredev")
  console.log(myArray)
