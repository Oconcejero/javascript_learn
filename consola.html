fround( )


  //Devuelve la representación flotante de precisión simple
  //más cercana de un número

  let numero = Math.fround(9.45215478542513265);
  console.log(numero) //9.452155113220215 redondea a 4 bytes

floor( )


  //Devuelve el número entero redondeado hacia abajo
  
  let numero = Math.floor(3.999);
  console.log(numero) //3

trunc( )


  //Devuelve la parte entera del número x, la eliminación de los dígitos fraccionarios
 
  let numero = Math.trunc(9.10);
  console.log(numero) //9 Elimina los decimales
