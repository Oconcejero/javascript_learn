MAPS

RETURN

Declaración


    let myMap = new Map()

    console.log(myMap)

Inicialización


    myMap = new Map([
        ["name", "Usuario"],
        ["email", "usuario@mail.com"],
        ["age", 37]
    ])

    console.log(myMap)

Métodos y propiedades


    // set - Actualizar elemento o agregarlo

    myMap.set("alias", "usuarioAlias") //Añade clave:valor
    myMap.set("name", "Usuario2") //Modifica la clave:valor

    console.log(myMap)

    // get - Para recuperar el valor

    console.log(myMap.get("name")) //Imprime el valor de la clave "name"
    console.log(myMap.get("surname")) //"Undefined" la clave no existe

    // has - Sirve para saber si existe o no dando true/false

    console.log(myMap.has("surname")) //false
    console.log(myMap.has("age")) //true

    // delete - Elimina por "clave"

    myMap.delete("email")

    console.log(myMap)

    // keys, values y entries

    console.log(myMap.keys()) //Retorna listado con las claves
    console.log(myMap.values()) //Retorna listado con los valores
    console.log(myMap.entries()) //Retorna los pares

    // size - Retorna el tamaño del mapa

    console.log(myMap.size)

    // clear - Para limpiar el mapa completo

    myMap.clear()

    console.log(myMap)
