Concatenación

    let myName = "Paco"
    let greeting = "Hola, " + myName + "!"
    console.log(greeting) // Hola, Paco!

    let cadena = "cadena de prueba";
    resultado = cadena.concat(" primera")

length( )

Devuelve el número de caracteres de la cadena


    let cadena1 = "cadena de prueba";
    console.log(cadena1.length) //16

startsWith( )

Si una cadena comienza con los caracteres de otra, devuelve true, sino, devuelve false


    let cadena1 = "cadena de prueba";
    let cadena2 = "cadena secundaria";
    let cadena3 = "esta es otra cadena de prueba";

    resultado = cadena1.startsWith(cadena2);
    console.log(resultado); //true
    resultado = cadena1. startsWith(cadena3);
    console.log(resultado); //false

endsWith( )

Si una cadena termina con los caracteres de otra, devuelve true, sino, devuelve false


    let cadena1 = "cadena de prueba";
    let cadena2 = "cadena secundaria";
    let cadena3 = "esta es otra cadena de prueba";

    resultado = cadena1.endsWith(cadena2);
    console.log(resultado); //false
    resultado = cadena3. endsWith(cadena1);
    console.log(resultado); //true

includes( )

Si una cadena puede encontrarse dentro de otro devuelve true, sino, devuelve false


    let cadena1 = "cadena de prueba";
    let cadena2 = "cadena secundaria";
    let cadena3 = "esta es otra cadena de prueba";

    resultado = cadena1.includes(cadena2);
    console.log(resultado); //false
    resultado = cadena3. includes(cadena1);
    console.log(resultado); //true

indexOf( )

Devuelve el índice del primer caracter de la cadena. Si no existe, devuelve -1


    let cadena1 = "cadena de prueba";

    resultado = cadena1.indexOf('prueba');
    console.log(resultado); //10, la posición de la primera letra

lastIndexOf( )

Devuelve el último índice del primer caracter de la cadena. Si no existe, devuelve -1


    let cadena1 = "cadena de prueba prueba prueba";

    resultado = cadena1.lastIndexOf('prueba');
    console.log(resultado); //24, la posición de la primera letra del último índice

padStart( ) & endStart( )

Rellena la cadena con los caracteres indicados.


    let cadena1 = "345";

    resultado = cadena1.padStart(5,"12");
    console.log(resultado); //12345
    resultado = cadena1.padEnd(5,"67");
    console.log(resultado); //34567

repeat( )

Devuelve la cadena repatida tantas veces se indique


    let cadena1 = "345 ";

    resultado = cadena1.repeat(5);
    console.log(resultado); //345 345 345 345 345
    resultado = cadena1.repeat(-1);
    console.log(resultado); //ERROR

split( )

Divide la cadena según se especifique


    let cadena1 = "Hola, como, estás";

    resultado = cadena1.split(",");
    console.log(resultado); //[ 'Hola', ' como', ' estás' ]

substring( )

Devuelve el pedazo de cadena que seleccionamos


    let cadena1 = "ABCDEFGHIJK";

    resultado = cadena1.substring(5,9);
    console.log(resultado); //FGHI

toLowerCase( )

Convierte una cadena a minúsculas


    let cadena1 = "HolA, como Estás";

    resultado = cadena1.toLowerCase();
    console.log(resultado); // hola, como estás

toUpperCase( )

Convierte una cadena a mayúsculas


    let cadena1 = "HolA, como Estás";

    resultado = cadena1.toLowerCase();
    console.log(resultado); // HOLA, COMO ESTÁS

toString( )

Devuelve una cadena que representa al objeto especificado


  let cadena1 = 50;

    resultado = cadena1.toString();
    console.log(resultado); //50
    console.log(typeof(resultado)); //string

trim( )

elimina los espacios en blandco al principio y al final de una cadena


  let cadena1 = "   Pedro   ";

    resultado = cadena1.trim();
    console.log(resultado); //"pedro"

trimEnd( ) & trimStart

Elimina los espacios en blanco al principio o al final de la cadena


  let cadena1 = "   Pedro   ";

    resultado = cadena1.trimEnd();
    console.log(resultado); //"   pedro"
    resultado = cadena1.trimStart();
    console.log(resultado); //"pedro   "
